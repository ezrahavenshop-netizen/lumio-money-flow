  transactions → admin transactions + user history
  admin_alerts → admin bell badge + alerts feed
  admin_settings → maintenance + settings sync

All subscriptions:

    { event: '*', schema: 'public',
      table: '[table]' },
    (payload) => { handle update }
  )
  .subscribe()

===================================================
ADDITIONAL SQL FOR SUPABASE
===================================================

Run in Supabase SQL Editor:

ALTER TABLE public.users
ADD COLUMN IF NOT EXISTS status text DEFAULT 'active';

ALTER TABLE public.transactions
ADD COLUMN IF NOT EXISTS user_id uuid REFERENCES
public.users(id) ON DELETE CASCADE;

CREATE INDEX IF NOT EXISTS transactions_user_id_idx
ON public.transactions(user_id);

CREATE INDEX IF NOT EXISTS transactions_created_at_idx
ON public.transactions(created_at DESC);
